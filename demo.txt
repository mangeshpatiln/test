// auth.js

class AuthService {
    constructor() {
        // Security Hotspot: Hardcoded credentials
        this.credentials = {
            username: "admin",
            password: "SuperSecret123!"  // <-- This will be flagged by Sonar
        };
    }

    login(inputUsername, inputPassword) {
        if (
            inputUsername === this.credentials.username &&
            inputPassword === this.credentials.password
        ) {
            console.log("Login successful!");
            return true;
        } else {
            console.log("Login failed.");
            return false;
        }
    }
}

// Example usage
const auth = new AuthService();
auth.login("admin", "SuperSecret123!");
